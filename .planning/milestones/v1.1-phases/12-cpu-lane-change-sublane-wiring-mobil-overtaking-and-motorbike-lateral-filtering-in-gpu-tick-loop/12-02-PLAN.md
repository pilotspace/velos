<objective>
Wire MOBIL lane-change overtaking (cars), motorbike sublane filtering, and PredictionService::update() into the GPU tick loop so lane-change decisions execute every frame and prediction overlay refreshes every 60 sim-seconds.

Purpose: Cars stuck behind slow leaders can overtake via MOBIL, motorbikes continuously adjust lateral position for filtering/weaving, and rerouting uses predicted future travel times instead of stale free-flow defaults. This completes the lane-change behavior pipeline and prediction loop that were built in earlier phases but never wired into tick_gpu().

Output: step_lane_changes() before GPU physics, step_motorbikes_sublane after GPU readback, step_prediction() after bus dwell — all in tick_gpu().
</objective>

       ```rust
       pub(crate) fn step_prediction(&mut self) {
           let prediction_service = match &mut self.reroute.prediction_service {
               Some(ps) => ps,
               None => return,
           };
           if !prediction_service.should_update(self.sim_time) {
               return;
           }
           // Gather edge flow/capacity data from road graph
           let edge_count = self.road_graph.inner().edge_count();
           let mut flows = vec![0.0_f32; edge_count];
           let mut capacities = vec![0.0_f32; edge_count];
           let mut free_flow = vec![0.0_f32; edge_count];
           let mut actual = vec![0.0_f32; edge_count];
           // Populate from road graph edge weights and agent counts per edge
           for edge_ref in self.road_graph.inner().edge_references() {
               let idx = edge_ref.id().index();
               if idx >= edge_count { continue; }
               let ew = edge_ref.weight();
               let speed_limit = ew.speed_limit_mps as f32;
               let length = ew.length_m as f32;
               capacities[idx] = (ew.lane_count as f32) * 40.0; // ~40 vehicles/lane capacity
               free_flow[idx] = if speed_limit > 0.0 { length / speed_limit } else { length / 13.9 };
               actual[idx] = free_flow[idx]; // default to free-flow, override below
           }
           // Count agents per edge for flow and compute actual travel times
           for (_entity, rp, kin) in self.world.query_mut::<(Entity, &RoadPosition, &Kinematics)>() {
               let idx = rp.edge_index as usize;
               if idx < edge_count {
                   flows[idx] += 1.0;
                   if kin.speed > 0.5 {
                       let edge_len = self.road_graph.inner().edge_weight(EdgeIndex::new(idx))
                           .map(|e| e.length_m as f32).unwrap_or(100.0);
                       actual[idx] = edge_len / (kin.speed as f32);
                   }
               }
           }
           let sim_hour = ((self.sim_time / 3600.0) as u8) % 24;
           let day_type = 0_u8; // weekday default
           let input = PredictionInput {
               flows: &flows,
               capacities: &capacities,
               free_flow: &free_flow,
               actual: &actual,
               hour: sim_hour,
               day_type,
           };
           prediction_service.update(&input, self.sim_time);
       }
       ```
